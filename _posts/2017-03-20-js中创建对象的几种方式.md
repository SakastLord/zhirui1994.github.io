# 1.使用对象字面量{}创建一个对象(推荐)
```javascript
var person = {};
    person.name = 'zhirui'; //对象属性
    person.age = 23;
    person.doSomething = function(){ //对象方法
        alert('Hello world!');
    }
```

# 2.new Object()创建对象
```javascript
var person = new Object();
    person.name = 'zhirui'; //对象属性
    person.age = 23;
    person.doSomething = function(){ //对象方法
        alert('Hello world!');
    }
```

# 3.工厂模式
```javascript
function createPerson(name, age, job){
    var o = new Object();
    o.name = name;
    o.age = age;
    o.job = job;
    o.doSomething = function(){
        alert('Hello world!');
    }
    
    return o;
};

var person = createPerson('zhirui', 23, 'frontEnd');
person.doSomething();
```
# 4.构造函数模式
```javascript
function Person(name, age, job) {
    this.name = name;
    this.age = age;
    this.job = job;
    this.doSomething = function(){
        alert('Hello world!');
    }
}

var person = new Person();
}
```

# 5.原型模式
```javascript
function Person(){};

Person.prototype.name = 'zhirui';
Person.prototype.age = 23;
Person.prototype.job = 'frontEnd';
Person.prototype.doSomething = function(){
        alert('Hello world!');
}
 
var person = new Person();
```

# 6.混合模式
```javascript
function Person(name, age, job){
    this.name = name;
    this.age = age;
    this.job = job;
}

Person.prototype = {
    constructor: Person,
    doSomething: function(){
        alert('Hello world!')
    }
}

var person = new Person('zhirui', 23, 'frontEnd');
```
